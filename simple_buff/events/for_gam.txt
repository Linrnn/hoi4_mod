country_event =
{
	id = for_gam.2
	title = "中立吧，冈比亚！"
	desc = "世界霸主冈比亚选择中立了？"
	is_triggered_only = yes
	option =
	{
		name = "中立"
		GAM =
		{
			dismantle_faction = yes
			create_faction = "第三世界"
			country_event = for_others.8
			country_event = for_gam.3
			set_politics =
			{
				ruling_party = neutrality		#执政党
				elections_allowed = no			#公众选举
			}
			add_popularity =
			{
				ideology = neutrality
				popularity = 1					#政党支持率
			}
			add_ideas = disarmed_nation			#征兵法案:非军事化国家
			add_ideas = without_reservation		#贸易法案:毫无保留
			add_ideas = undisturbed_isolation	#经济法案:不受干扰的孤立
		}
	}
	option =
	{
		name = "继续"
		GAM = 
		{
			country_event = for_others.8
			country_event = for_gam.3
		}
	}
	option = { name = "退出" }
}
